float calculateUt(float yt, ofstream &file) {
    float q0 = K * (1.0 + (TD / T0));
    float q1 = -K * (1.0 + (2.0 * TD / T0) - (T0 / T));
    float q2 = K * (TD / T0);
    file << "," << ek << endl;
    ek = wt - yt;
    ut += (q0 * ek) + (q1 * ek1) + (q2 * ek2);
    ek2 = ek1;
    ek1 = ek;
    return ut;
}
